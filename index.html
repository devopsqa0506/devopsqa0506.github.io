// SITEBOOK APPS SCRIPT v7 // action comes from URL param (?action=xxx) — survives 302 redirect // data comes from POST body (no-cors) — delivered fine // GET used only for readAll and getConfig (read operations) function doGet(e) { var action = e.parameter.action; if (action === 'getConfig') { try { var files = DriveApp.getFilesByName('SiteBook_Config.json'); if (!files.hasNext()) return out({ok:false, error:'SiteBook_Config.json not found in Drive root'}); var cfg = JSON.parse(files.next().getBlob().getDataAsString()); return out({ok:true, data:cfg}); } catch(err) { return out({ok:false, error:err.message}); } } if (action === 'readAll') { try { var ss = SpreadsheetApp.openById(e.parameter.sheetId); var sheets = e.parameter.sheets ? JSON.parse(e.parameter.sheets) : ['Entries','Vendors','Prices','Documents','Contacts','Projects','BillHeaders','BillItems','Loans','LoanPayments','ProjectIncome','ProjectExpenses','LandPayments','PlotLandPayments']; var result = {}; sheets.forEach(function(name) { try { var ws = ss.getSheetByName(name); result[name] = ws ? ws.getDataRange().getValues() : []; } catch(e) { result[name] = []; } }); return out({ok:true, data:result}); } catch(err) { return out({ok:false, error:err.message}); } } return out({ok:false, error:'Unknown GET action: ' + action}); } function doPost(e) { // action from URL param — survives 302 redirect var action = e.parameter.action; var req = {}; try { req = JSON.parse(e.postData.contents); } catch(err) { return out({ok:false, error:'Invalid JSON: '+err.message}); } var ss = SpreadsheetApp.openById(req.sheetId); if (action === 'append') { try { var ws = ss.getSheetByName(req.sheetName); if (!ws) { ws = ss.insertSheet(req.sheetName); ws.appendRow(req.headers); } ws.appendRow(req.row); return out({ok:true}); } catch(err) { return out({ok:false, error:err.message}); } } if (action === 'deleteRow') { try { var ws = ss.getSheetByName(req.sheetName); if (!ws) return out({ok:false, error:'Sheet not found: '+req.sheetName}); var data = ws.getDataRange().getValues(); var col = data[0].indexOf(req.matchCol); if (col === -1) return out({ok:false, error:'Column not found: '+req.matchCol}); var idCol = data[0].indexOf('id'); var excludePrefix = req.excludeIdPrefix || null; for (var i = data.length-1; i >= 1; i--) { if (String(data[i][col]) === String(req.matchVal)) { if (excludePrefix && idCol !== -1 && String(data[i][idCol]).indexOf(excludePrefix) === 0) continue; ws.deleteRow(i+1); } } return out({ok:true}); } catch(err) { return out({ok:false, error:err.message}); } } if (action === 'batchAppend') { try { var ws = ss.getSheetByName(req.sheetName); if (!ws) { ws = ss.insertSheet(req.sheetName); ws.appendRow(req.headers); } (req.rows||[]).forEach(function(row) { ws.appendRow(row); }); return out({ok:true}); } catch(err) { return out({ok:false, error:err.message}); } } if (action === 'updateRow') { try { var ws = ss.getSheetByName(req.sheetName); if (!ws) { ws = ss.insertSheet(req.sheetName); ws.appendRow(req.headers); } var data = ws.getDataRange().getValues(); var col = data[0].indexOf(req.matchCol); if (col === -1) return out({ok:false, error:'Column not found: '+req.matchCol}); var found = false; for (var i = 1; i < data.length; i++) { if (data[i][col] == req.matchVal) { for (var j = 0; j < req.row.length; j++) ws.getRange(i+1, j+1).setValue(req.row[j]); found = true; break; } } if (!found) ws.appendRow(req.row); return out({ok:true, found:found}); } catch(err) { return out({ok:false, error:err.message}); } } if (action === 'saveFile') { try { var folderName = req.folder || 'SiteBook Documents'; var folders = DriveApp.getFoldersByName(folderName); var folder = folders.hasNext() ? folders.next() : DriveApp.createFolder(folderName); var blob = Utilities.newBlob(Utilities.base64Decode(req.fileData), req.mimeType||'application/octet-stream', req.fileName||'document'); var file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return out({ok:true, fileId:file.getId(), url:file.getUrl()}); } catch(err) { return out({ok:false, error:err.message}); } } return out({ok:false, error:'Unknown POST action: ' + action}); } function out(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }
